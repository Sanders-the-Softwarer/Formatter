-- Комментарии, а также проверка выравнивания в комментариях
comment on table v$dual is 'Вьюха';
comment on column v$dual.id is 'Ключ';
comment on column v$dual.name is 'Название';

-- Можно создавать таблицы
create table x (i integer) tablespace tmp

-- Таблицы могут быть индекс-организованными
create table xi (i integer) organization index tablespace tmp

-- Таблицы могут быть временными
create global temporary table t (i integer) on commit delete rows
create global temporary table tt (ii integer) on commit preserve rows

-- Поля таблиц могут содержать разные типы, default и not null
create table x(id integer not null, di varchar2(2000 char) default 'abc' || 'de', ididi date default sysdate not null)

-- В таблицах могут быть указаны все типы constraint-ов и поля вперемешку
create table x (
  i integer,
  constraint pk primary key(i) using index tablespace idx,
  j varchar2(100 byte) default null,
  constraint fk foreign key (i, j) references y (ii, jj),
  k date not null,
  constraint uq unique (k) using index tablespace idx,
  l some.tipe,
  constraint chk check (1 + 2 = 3)
) tablespace tbl

-- В таблицах можно указывать место хранения lob-ов
create table x (
  i blob,
  j blob
) tablespace t1
lob(i) store as x_i(tablespace t2)
lob(j) store as x_j(tablespace t3);

-- можно создавать последовательности
create sequence seq
    minvalue 1
    maxvalue 999999999999999999999999999
    start with 100000
    increment by 1
    order
    nocycle
    nocache;
